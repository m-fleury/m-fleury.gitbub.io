  is [] ‹uncurry lbd_ref_write›
  :: ‹ [λ(lbd, i). i ≤ Suc (uint32_max div 2)]⇩a
     lbd_int_assn⇧d *⇩a uint32_nat_assn⇧k → lbd_int_assn›

  apply (annot_unat_const ‹TYPE(32)›)
  by sepref
